6) Create the brain node file
A) Create brain_node.py
nano ~/ros2_ws/src/brain_pkg/brain_pkg/brain_node.py


Paste this minimal working node (this is basically what we tested):

import rclpy
from rclpy.node import Node
from std_msgs.msg import Bool, Float32, String

class BrainNode(Node):
    def __init__(self):
        super().__init__('brain_node')

        self.person_detected = False
        self.person_distance = 999.0  # default far away

        self.create_subscription(Bool, '/person_detected', self.person_cb, 10)
        self.create_subscription(Float32, '/person_distance', self.dist_cb, 10)

        self.cmd_pub = self.create_publisher(String, '/vehicle_cmd', 10)

        self.timer = self.create_timer(0.5, self.tick)

    def person_cb(self, msg: Bool):
        self.person_detected = msg.data

    def dist_cb(self, msg: Float32):
        self.person_distance = msg.data

    def tick(self):
        # Simple rule:
        if self.person_detected and self.person_distance < 2.0:
            cmd = "STOP"
        else:
            cmd = "GO"

        out = String()
        out.data = cmd
        self.cmd_pub.publish(out)
        self.get_logger().info(f"Decision: {cmd} (detected={self.person_detected}, dist={self.person_distance:.2f})")

def main(args=None):
    rclpy.init(args=args)
    node = BrainNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()

13) Windows publisher test (no camera)

Create publish_test.py on Windows:

import time
import roslibpy

client = roslibpy.Ros(host='localhost', port=9090)
client.run()

person_pub = roslibpy.Topic(client, '/person_detected', 'std_msgs/Bool')
dist_pub   = roslibpy.Topic(client, '/person_distance', 'std_msgs/Float32')

try:
    while True:
        person_pub.publish(roslibpy.Message({'data': True}))
        dist_pub.publish(roslibpy.Message({'data': 1.5}))
        time.sleep(2)

        dist_pub.publish(roslibpy.Message({'data': 3.0}))
        time.sleep(2)

        person_pub.publish(roslibpy.Message({'data': False}))
        time.sleep(2)

finally:
    person_pub.unadvertise()
    dist_pub.unadvertise()
    client.terminate()


Run it:

python publish_test.py


Your WSL brain should flip STOP/GO.

If localhost fails:

In WSL run:

hostname -I


Then set host in Windows to that IP:

client = roslibpy.Ros(host='172.xx.xx.xx', port=9090)
